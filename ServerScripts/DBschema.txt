CREATE TABLE PROFILES IF NOT EXISTS(
	EMAIL	VARCHAR(30) 	NOT NULL	UNIQUE,
	USERNAME VARCHAR(15)	NOT NULL	UNIQUE,
	BIO		VARCHAR(200),
	IMAGE PHOTO,
	PRIMARY KEY(email)
)

CREATE TABLE COURSES IF NOT EXISTS(
	COURSEID	VARCHAR(64)	NOT NULL,
	INSTRUCTOR	VARCHAR(64),
	TERM		NOT NULL,
	COURSENAME	NOT NULL,
	PRIMARY KEY (COURSEID,TERM)
)

CREATE TABLE MESSAGES IF NOT EXISTS(
	MSGID	VARCHAR(64)		NOT NULL	UNIQUE,
	MSGTEXT	VARCHAR(500)	NOT NULL,
	COURSEID	VARCHAR(64)		NOT NULL,
	IMAGE PHOTO,
	LINK	VARCHAR(200),
	PRIMARY KEY(MSGID),
	FOREIGN KEY(COURSEID) REFERENCES COURSES(COURSEID)
)

CREATE TABLE ENROLLMENT IF NOT EXISTS(
	EMAIL	VARCHAR(30) NOT NULL,
	COURSEID VARCHAR(30)	NOT NULL,
	PRIMARY KEY(EMAIL,COURSEID),
	FOREIGN KEY(EMAIL) REFERENCES PROFILE(EMAIL),
	FOREIGN KEY(COURSEID) REFERENCES COURSES(COURSEID)
)
